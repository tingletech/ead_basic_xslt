def a2html(index):
    ret = "<div>\n"
    for i in index:
        ret += '  <div><a href="%s">%s</a></div>\n' % (i, i)
    ret += "</div>\n"
    return ret

def mkdir_p(path):

    except OSError as exc: # Python >2.5
        if exc.errno == errno.EEXIST:
            pass
